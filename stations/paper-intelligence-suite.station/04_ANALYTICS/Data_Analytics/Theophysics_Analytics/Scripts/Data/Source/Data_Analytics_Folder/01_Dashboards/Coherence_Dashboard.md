## Results

```dataviewjs
// This script reads the JSON file generated by our Python script.
const path = "Data_Analytics_Folder/01_Dashboards/coherence_results.json";

// Read the file content. dv.io.load() is an asynchronous function.
const content = await dv.io.load(path);

// Check if the content was loaded successfully
if (content) {
    // Parse the JSON content
    const data = JSON.parse(content);

    // Display the results using Dataview's rendering components
    dv.header(2, "Latest Analysis");
    dv.paragraph(`**Coherence Score:** ${data.coherence_score}`);
    dv.paragraph(`**Last Run:** ${data.last_run}`);
    
    dv.header(3, "Files Analyzed");
    dv.list(data.analyzed_files);

} else {
    dv.paragraph("Could not load analysis results. Run the script to generate the data file.");
}
```
